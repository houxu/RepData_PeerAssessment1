Are there differences in activity patterns between weekdays and weekends?

day_type <- function(date) {
  if (weekdays(date) %in% c("星期六", "星期日")) {
    return ("weekend")
  } else {
    return ("weekday")
  }
}
data_no_NA$day_type <- as.factor(sapply(data_no_NA$date, day_type))
